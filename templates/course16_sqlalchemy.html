Урок 1: Введение в SQLAlchemy и Core

Что такое SQLAlchemy?

SQLAlchemy — это мощная библиотека Python для работы с реляционными базами данных. Она предоставляет два основных подхода: Core (низкоуровневый SQL-выражения) и ORM (объектно-реляционное отображение).

# Установка: pip install sqlalchemy
from sqlalchemy import create_engine, text

# Создание подключения (SQLite для примера)
engine = create_engine("sqlite:///example.db", echo=True)

# Выполнение raw SQL через Core
with engine.connect() as conn:
    result = conn.execute(text("SELECT 'Hello, DB'"))
    print(result.all())

Движок и сессии

Engine — точка входа в базу. Session — рабочая область ORM. Core использует подключения напрямую.

from sqlalchemy import MetaData, Table, Column, Integer, String

users = Table(
    'users', metadata,
    Column('id', Integer, primary_key=True),
    Column('name', String(50)),
    Column('age', Integer)

Урок 2: Модели и базовые операции ORM

Declarative Base

from sqlalchemy.orm import declarative_base, sessionmaker

Base = declarative_base()

class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True)
    name = Column(String(50), nullable=False)
    age = Column(Integer)

Base.metadata.create_all(engine)
Session = sessionmaker(bind=engine)
session = Session()

# CREATE
new_user = User(name="Анна", age=28)
session.add(new_user)
session.commit()

# READ
user = session.query(User).filter_by(name="Анна").first()
print(user.name, user.age)

# UPDATE
user.age = 29
session.commit()

# DELETE
session.delete(user)
session.commit()

Урок 3: Отношения

Один-ко-многим (ForeignKey)

from sqlalchemy import ForeignKey
from sqlalchemy.orm import relationship

class Post(Base):
    __tablename__ = 'posts'
    id = Column(Integer, primary_key=True)
    title = Column(String)
    user_id = Column(Integer, ForeignKey('users.id'))
    author = relationship("User", back_populates="posts")

User.posts = relationship("Post", order_by=Post.id, back_populates="author")

user = session.query(User).first()
new_post = Post(title="Мой первый пост", author=user)
session.add(new_post)
session.commit()
print(user.posts[0].title)

Многие-ко-многим

association_table = Table('user_group', Base.metadata,
    Column('user_id', ForeignKey('users.id')),
    Column('group_id', ForeignKey('groups.id'))
)
class Group(Base):
    __tablename__ = 'groups'
    id = Column(Integer, primary_key=True)
    name = Column(String)
    members = relationship("User", secondary=association_table, back_populates="groups")

User.groups = relationship("Group", secondary=association_table, back_populates="members")

Урок 5: Оптимизация запросов

joinedload и selectinload

from sqlalchemy.orm import joinedload, selectinload

# Ленивая загрузка (по умолчанию) – N+1 проблема
users = session.query(User).all()
for u in users:
    print(u.posts)  # выполнит отдельный запрос для каждого юзера

# Жадная загрузка (JOIN)
users = session.query(User).options(joinedload(User.posts)).all()
# selectinload – отдельный запрос, но эффективнее для больших данных
users = session.query(User).options(selectinload(User.posts)).all()
# Пагинация и индексы
users = session.query(User).filter(User.age > 18).limit(10).offset(0).all()
# Индексы: Index('idx_user_age', User.age)

Урок 6: Асинхронный SQLAlchemy

asyncpg + SQLAlchemy 1.4+

from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
from sqlalchemy.orm import sessionmaker

engine = create_async_engine("postgresql+asyncpg://user:pass@localhost/db")
async_session = sessionmaker(engine, class_=AsyncSession)

async def get_users():
    async with async_session() as session:
        result = await session.execute(select(User))
        users = result.scalars().all()
        return users

    post_id = Column(Integer, ForeignKey('posts.id'))
