Урок 1: Введение в асинхронность и event loop

Что такое асинхронность?

Асинхронное программирование позволяет выполнять несколько задач одновременно, не блокируя выполнение основной программы. В Python для этого используется модуль asyncio.

Синхронный vs Асинхронный подход:

import asyncio
import time

# Синхронный подход (блокирующий)
def sync_sleep(seconds):
    print(f"Синхронный сон на {seconds} сек")
    time.sleep(seconds)
    print("Проснулись!")

# Асинхронный подход (неблокирующий)
async def async_sleep(seconds):
    print(f"Асинхронный сон на {seconds} сек")
    await asyncio.sleep(seconds)
    print("Проснулись!")

# Синхронное выполнение
start = time.time()
sync_sleep(2)
sync_sleep(2)
print(f"Синхронно: {time.time() - start:.2f} сек")

# Асинхронное выполнение
async def async_main():
    start = time.time()
    await asyncio.gather(async_sleep(2), async_sleep(2))
    print(f"Асинхронно: {time.time() - start:.2f} сек")

# asyncio.run(async_main())

Event Loop (цикл событий)

Event loop — это сердце асинхронного программирования. Он управляет выполнением корутин, планирует задачи и обрабатывает события.

import asyncio

async def task(name, delay):
    print(f"Задача {name} началась")
    await asyncio.sleep(delay)
    print(f"Задача {name} завершилась")
    return f"Результат {name}"

async def main():
    # Получаем текущий event loop
    loop = asyncio.get_running_loop()
    print(f"Event loop: {loop}")
    
    # Создаем задачи
    task1 = loop.create_task(task("A", 2))
    task2 = loop.create_task(task("B", 1))
    
    # Ждем завершения
    results = await asyncio.gather(task1, task2)
    print(results)

# Запускаем event loop
asyncio.run(main())

Важно: Event loop работает в одном потоке, но может переключаться между задачами во время ожидания (await). Это позволяет эффективно использовать время при I/O операциях.

Урок 2: Корутины и задачи

Корутины (coroutines)

Корутина — это функция, объявленная с async def. Она может приостанавливать свое выполнение с помощью await и возвращать управление event loop.

async def fetch_data(url):
    """Корутина для получения данных"""
    print(f"Загрузка {url}...")
    await asyncio.sleep(1)  # Имитация сетевого запроса
    return f"Данные из {url}"

async def process_data(data):
    """Корутина для обработки данных"""
    await asyncio.sleep(0.5)  # Имитация обработки
    return f"Обработано: {data}"

async def main():
    # Вызов корутины не запускает ее сразу
    data_task = fetch_data("api.example.com")
    
    # Чтобы запустить, нужно использовать await
    data = await data_task
    result = await process_data(data)
    print(result)

# asyncio.run(main())

Задачи (Tasks)

Задача — это обертка для корутины, которая позволяет запустить ее конкурентно. Задачи планируются event loop и могут выполняться параллельно.

async def worker(name, delay):
    for i in range(3):
        print(f"{name}: шаг {i}")
        await asyncio.sleep(delay)
    return f"{name} завершил работу"

async def main():
    # Создаем задачи (запускаем конкурентно)
    task1 = asyncio.create_task(worker("Алиса", 0.5))
    task2 = asyncio.create_task(worker("Боб", 0.7))
    task3 = asyncio.create_task(worker("Чарли", 0.3))
    
    # Ждем завершения всех задач
    results = await asyncio.gather(task1, task2, task3)
    print(results)
    
    # Проверяем статус задач
    for task in [task1, task2, task3]:
        print(f"{task.get_name()}: done={task.done()}")

asyncio.run(main())

Совет: Используйте asyncio.gather() для одновременного запуска нескольких задач, и asyncio.wait() для более гибкого управления.

Урок 3: Асинхронные примитивы синхронизации

Lock и Semaphore

Для защиты общего ресурса в асинхронном коде используются специальные примитивы.

import asyncio

# Асинхронный замок (Lock)
async def worker_lock(lock, name):
    async with lock:
        print(f"{name} захватил замок")
        await asyncio.sleep(1)
        print(f"{name} освободил замок")

# Семафор (ограничивает количество одновременных задач)
async def worker_semaphore(semaphore, name):
    async with semaphore:
        print(f"{name} начал работу (одновременно может быть {semaphore._value})")
        await asyncio.sleep(1)
        print(f"{name} закончил работу")

async def main():
    # Пример с Lock
    lock = asyncio.Lock()
    await asyncio.gather(
        worker_lock(lock, "A"),
        worker_lock(lock, "B"),
        worker_lock(lock, "C")
    )
    
    # Пример с Semaphore (максимум 2 одновременно)
    semaphore = asyncio.Semaphore(2)
    await asyncio.gather(
        worker_semaphore(semaphore, "1"),
        worker_semaphore(semaphore, "2"),
        worker_semaphore(semaphore, "3"),
        worker_semaphore(semaphore, "4")
    )

# asyncio.run(main())

Event и Queue

import asyncio

# Event — для уведомлений между задачами
async def waiter(event, name):
    print(f"{name} ждет события...")
    await event.wait()
    print(f"{name} получил событие!")

async def setter(event):
    await asyncio.sleep(2)
    print("Устанавливаю событие!")
    event.set()

# Queue — очередь задач (производитель-потребитель)
async def producer(queue, name):
    for i in range(5):
        item = f"Элемент {i} от {name}"
        await asyncio.sleep(0.5)
        await queue.put(item)
        print(f"[{name}] Создал: {item}")
    await queue.put(None)  # Сигнал завершения

async def consumer(queue, name):
    while True:
        item = await queue.get()
        if item is None:
            break
        await asyncio.sleep(0.2)
        print(f"[{name}] Обработал: {item}")

async def main_queue():
    queue = asyncio.Queue(maxsize=3)
    producers = [asyncio.create_task(producer(queue, f"P{i}")) for i in range(2)]
    consumers = [asyncio.create_task(consumer(queue, f"C{i}")) for i in range(3)]
    
    await asyncio.gather(*producers)
    await queue.join()
    for c in consumers:
        c.cancel()

# asyncio.run(main_queue())

Урок 4: Асинхронные HTTP-запросы

Библиотека aiohttp

aiohttp — основная библиотека для асинхронных HTTP-запросов в Python.

Установка:

pip install aiohttp

Пример: параллельные запросы

import asyncio
import aiohttp

urls = [
    "https://jsonplaceholder.typicode.com/posts/1",
    "https://jsonplaceholder.typicode.com/posts/2",
    "https://jsonplaceholder.typicode.com/posts/3",
]

async def fetch_url(session, url):
    async with session.get(url) as response:
        return await response.json()

async def main():
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
        
        for result in results:
            print(f"ID: {result['id']}, Заголовок: {result['title'][:50]}...")

# asyncio.run(main())

Ограничение параллелизма

import asyncio
import aiohttp

async def fetch_with_semaphore(session, url, semaphore):
    async with semaphore:
        async with session.get(url) as response:
            return await response.text()

async def main():
    urls = [f"https://jsonplaceholder.typicode.com/posts/{i}" for i in range(1, 21)]
    semaphore = asyncio.Semaphore(5)  # Максимум 5 одновременных запросов
    
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_with_semaphore(session, url, semaphore) for url in urls]
        results = await asyncio.gather(*tasks)
        print(f"Загружено {len(results)} страниц")

# asyncio.run(main())

Важно: Всегда используйте ClientSession как контекстный менеджер. Это обеспечивает правильное управление соединениями.

Урок 5: Асинхронные драйверы баз данных

asyncpg — асинхронный PostgreSQL

asyncpg — самый быстрый асинхронный драйвер для PostgreSQL.

Установка:

pip install asyncpg
import asyncio
import asyncpg

async def main():
    # Подключение к базе данных
    conn = await asyncpg.connect(
        user='user',
        password='password',
        database='test',
        host='localhost'
    )
    
    try:
        # Создание таблицы
        await conn.execute('''
            CREATE TABLE IF NOT EXISTS users (
                id SERIAL PRIMARY KEY,
                name TEXT,
                age INTEGER
            )
        ''')
        
        # Вставка данных
        await conn.execute(
            'INSERT INTO users(name, age) VALUES($1, $2)',
            'Анна', 25
        )
        
        # Запрос данных
        rows = await conn.fetch('SELECT * FROM users')
        for row in rows:
            print(f"ID: {row['id']}, Имя: {row['name']}, Возраст: {row['age']}")
        
        # Параллельные запросы
        async with conn.transaction():
            tasks = [
                conn.fetchval('SELECT COUNT(*) FROM users'),
                conn.fetchval('SELECT AVG(age) FROM users'),
                conn.fetchval('SELECT MAX(age) FROM users')
            ]
            count, avg_age, max_age = await asyncio.gather(*tasks)
            print(f"Всего: {count}, Средний возраст: {avg_age}, Максимальный: {max_age}")
            
    finally:
        await conn.close()

# asyncio.run(main())

FastAPI — современный фреймворк

FastAPI построен на asyncio и предоставляет невероятную производительность.

from fastapi import FastAPI
import asyncio
import asyncpg

app = FastAPI()

@app.get("/")
async def root():
    return {"message": "Hello World"}

@app.get("/users/{user_id}")
async def get_user(user_id: int):
    # Асинхронный запрос к БД
    conn = await asyncpg.connect('postgresql://user:pass@localhost/db')
    user = await conn.fetchrow('SELECT * FROM users WHERE id = $1', user_id)
    await conn.close()
    return user

aiohttp сервер

from aiohttp import web
import asyncio

async def handle(request):
    name = request.match_info.get('name', "Anonymous")
    text = f"Hello, {name}"
    return web.Response(text=text)

async def main():
    app = web.Application()
    app.add_routes([web.get('/', handle),
                    web.get('/{name}', handle)])
    
    runner = web.AppRunner(app)
    await runner.setup()
    site = web.TCPSite(runner, 'localhost', 8080)
    await site.start()
    print("Сервер запущен на http://localhost:8080")
    
    # Держим сервер запущенным
    await asyncio.Event().wait()

Урок 8: Практический проект: Асинхронный парсер

Проект: Асинхронный парсер новостей

Создадим полноценный асинхронный парсер, который загружает данные с нескольких сайтов, обрабатывает их и сохраняет в базу данных.

import asyncio
import aiohttp
import asyncpg
from bs4 import BeautifulSoup
import time

class AsyncNewsParser:
    def __init__(self, db_pool, max_concurrent=5):
        self.pool = db_pool
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.results = []
    
    async def fetch_page(self, session, url):
        async with self.semaphore:
            try:
                async with session.get(url, timeout=10) as response:
                    return url, await response.text()
            except Exception as e:
                print(f"Ошибка загрузки {url}: {e}")
                return url, None
    
    async def parse_page(self, url, html):
        if not html:
            return None
        
        # CPU-bound парсинг в отдельном потоке
        loop = asyncio.get_event_loop()
        soup = await loop.run_in_executor(None, BeautifulSoup, html, 'html.parser')
        
        titles = []
        for tag in soup.find_all(['h2', 'h3'])[:5]:
            text = tag.get_text().strip()
            if text and len(text) > 10:
                titles.append(text[:100])
        
        return {
            'url': url,
            'titles': titles,
            'timestamp': time.time()
        }
    
    async def save_to_db(self, data):
        if not data:
            return
        
        async with self.pool.acquire() as conn:
            await conn.execute('''
                INSERT INTO parsed_data (url, titles, timestamp)
                VALUES ($1, $2, $3)
            ''', data['url'], data['titles'], data['timestamp'])
    
    async def process_site(self, session, url):
        url, html = await self.fetch_page(session, url)
        result = await self.parse_page(url, html)
        if result:
            await self.save_to_db(result)
            print(f"✓ Обработан: {url} (найдено {len(result['titles'])} заголовков)")
    
    async def run(self, urls):
        async with aiohttp.ClientSession() as session:
            tasks = [self.process_site(session, url) for url in urls]
            await asyncio.gather(*tasks)

async def init_db():
    # Создание таблицы
    conn = await asyncpg.connect('postgresql://user:pass@localhost/news')
    await conn.execute('''
        CREATE TABLE IF NOT EXISTS parsed_data (
            id SERIAL PRIMARY KEY,
            url TEXT,
            titles TEXT[],
            timestamp FLOAT
        )
    ''')
    await conn.close()
    return await asyncpg.create_pool('postgresql://user:pass@localhost/news', min_size=5, max_size=20)

async def main():
    urls = [
        'https://news.ycombinator.com',
        'https://www.python.org/blogs/',
        'https://realpython.com/',
        'https://dev.to/t/python',
        'https://habr.com/ru/hub/python/',
    ]
    
    pool = await init_db()
    parser = AsyncNewsParser(pool, max_concurrent=3)
    
    start = time.time()
    await parser.run(urls)
    elapsed = time.time() - start
    
    print(f"\nПарсинг завершен за {elapsed:.2f} секунд")
    await pool.close()

if __name__ == "__main__":
    asyncio.run(main())

Этот проект демонстрирует:

  • Асинхронные HTTP-запросы с aiohttp
  • Ограничение конкурентности через Semaphore
  • Работу с асинхронной базой данных (asyncpg)
  • Сочетание CPU-bound (парсинг) с asyncio
  • Пул соединений для базы данных
