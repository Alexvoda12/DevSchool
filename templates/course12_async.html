Урок 1: Синхронное vs Асинхронное программирование

Синхронный код (обычный)

В синхронном программировании операции выполняются последовательно. Пока одна операция не закончится, следующая не начнется.

import time

def task(name, seconds):
    print(f"Задача {name} началась")
    time.sleep(seconds)  # Имитация долгой операции (I/O, запрос к API и т.д.)
    print(f"Задача {name} завершилась")
    return f"Результат {name}"

# Синхронное выполнение
start = time.time()

result1 = task("A", 2)
result2 = task("B", 3)
result3 = task("C", 1)

print(f"Все результаты: {result1}, {result2}, {result3}")
print(f"Общее время: {time.time() - start:.2f} секунд")
# Общее время: ~6 секунд (2+3+1)

Проблема: программа простаивает, пока ждет завершения I/O операций (чтение файлов, запросы к сети, работа с БД).

Асинхронный код

Асинхронный код позволяет программе не ждать завершения операции, а переключиться на другую задачу.

import asyncio
import time

async def task(name, seconds):
    print(f"Задача {name} началась")
    await asyncio.sleep(seconds)  # Асинхронное ожидание
    print(f"Задача {name} завершилась")
    return f"Результат {name}"

async def main():
    # Запускаем задачи конкурентно
    task_a = asyncio.create_task(task("A", 2))
    task_b = asyncio.create_task(task("B", 3))
    task_c = asyncio.create_task(task("C", 1))
    
    # Ждем завершения всех задач
    results = await asyncio.gather(task_a, task_b, task_c)
    print(f"Все результаты: {results}")

start = time.time()
asyncio.run(main())
print(f"Общее время: {time.time() - start:.2f} секунд")
# Общее время: ~3 секунды (максимальное время из всех задач)

Аналогия из жизни:

Синхронно: Вы заказываете кофе и стоите у стойки, пока его готовят, ничего больше не делая.

Асинхронно: Вы заказываете кофе, берете номерок и идете читать книгу. Когда кофе готов, вам приносят его.

Сравнение подходов

Характеристика Синхронный Асинхронный
Выполнение Последовательное Конкурентное (переключение)
Ожидание I/O Блокирует программу Не блокирует, переключается
Сложность кода Низкая Выше (async/await)
Производительность Ниже при I/O операциях Выше при I/O операциях
Для CPU-bound задач Подходит Не подходит

CPU-bound задачи: требуют много вычислений (сортировка, математика) — здесь асинхронность не поможет.

I/O-bound задачи: ожидание ввода-вывода (сеть, диски) — здесь асинхронность дает огромный прирост.

Урок 2: Основы asyncio и корутины

Ключевые понятия

  • Корутина (coroutine): функция, объявленная с async def. Может приостанавливать выполнение.
  • await: ключевое слово для ожидания результата корутины.
  • Событийный цикл (event loop): сердце asyncio, управляет выполнением задач.
  • Задача (Task): обертка для корутины, позволяющая запустить её конкурентно.
import asyncio

# Определяем корутину
async def say_hello():
    print("Привет")
    await asyncio.sleep(1)  # Асинхронное ожидание
    print("Мир")

# Запуск корутины
asyncio.run(say_hello())

# Более сложный пример
async def fetch_data(seconds, name):
    print(f"Начинаем загрузку {name}...")
    await asyncio.sleep(seconds)  # Имитация загрузки данных
    print(f"Загрузка {name} завершена")
    return f"Данные {name}"

async def main():
    # Последовательное выполнение
    result1 = await fetch_data(2, "A")
    result2 = await fetch_data(1, "B")
    print(result1, result2)

# asyncio.run(main())  # Займет 3 секунды

async def main_concurrent():
    # Конкурентное выполнение
    task1 = asyncio.create_task(fetch_data(2, "A"))
    task2 = asyncio.create_task(fetch_data(1, "B"))
    
    result1 = await task1
    result2 = await task2
    print(result1, result2)

# asyncio.run(main_concurrent())  # Займет 2 секунды (максимальное время)

Создание и управление задачами

import asyncio

async def worker(name, delay):
    for i in range(3):
        print(f"Работник {name}: шаг {i}")
        await asyncio.sleep(delay)
    return f"Работник {name} завершил работу"

async def main():
    # Создаем несколько задач
    tasks = [
        asyncio.create_task(worker("A", 1), name="WorkerA"),
        asyncio.create_task(worker("B", 1.5), name="WorkerB"),
        asyncio.create_task(worker("C", 0.5), name="WorkerC"),
    ]
    
    # Ждем все задачи
    results = await asyncio.gather(*tasks)
    print(f"Результаты: {results}")
    
    # Информация о задачах
    for task in tasks:
        print(f"Задача {task.get_name()}: завершена={task.done()}")

asyncio.run(main())

asyncio.gather vs asyncio.wait

import asyncio

async def task(name, delay, fail=False):
    await asyncio.sleep(delay)
    if fail:
        raise ValueError(f"Ошибка в задаче {name}")
    return f"Результат {name}"

async def main():
    # gather - собирает результаты всех задач
    try:
        results = await asyncio.gather(
            task("A", 1),
            task("B", 2),
            task("C", 0.5),
            return_exceptions=False  # Если True, возвращает исключения как результаты
        )
        print(f"Gather results: {results}")
    except Exception as e:
        print(f"Gather поймал ошибку: {e}")
    
    # wait - ожидает задачи с определенными условиями
    tasks = [
        asyncio.create_task(task("X", 1)),
        asyncio.create_task(task("Y", 2)),
        asyncio.create_task(task("Z", 3)),
    ]
    
    done, pending = await asyncio.wait(tasks, timeout=2.5)
    
    print(f"Завершено: {len(done)} задач")
    for t in done:
        print(f"  {t.result()}")
    
    print(f"Ожидают: {len(pending)} задач")
    # Отменяем оставшиеся задачи
    for t in pending:
        t.cancel()

asyncio.run(main())

Важно! В асинхронном коде нельзя использовать обычный time.sleep() — он заблокирует весь событийный цикл. Всегда используйте await asyncio.sleep().

Пример: асинхронный aiohttp (быстро)

import asyncio
import aiohttp
import time

urls = [
    "https://jsonplaceholder.typicode.com/posts/1",
    "https://jsonplaceholder.typicode.com/posts/2",
    "https://jsonplaceholder.typicode.com/posts/3",
    "https://jsonplaceholder.typicode.com/posts/4",
    "https://jsonplaceholder.typicode.com/posts/5",
]

async def fetch_url(session, url):
    async with session.get(url) as response:
        return await response.json()

async def fetch_all():
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
        return results

start = time.time()
results = asyncio.run(fetch_all())
print(f"Время: {time.time() - start:.2f} сек")
# ~0.5-1 секунда (все запросы параллельно)

Ограничение количества одновременных запросов

Чтобы не перегружать сервер, можно использовать семафоры для ограничения конкурентности.

import asyncio
import aiohttp

async def fetch_url(session, url, semaphore):
    async with semaphore:  # Ограничиваем количество одновременных запросов
        print(f"Запрос к {url}")
        async with session.get(url) as response:
            return await response.text()

async def main():
    urls = [f"https://jsonplaceholder.typicode.com/posts/{i}" for i in range(1, 21)]
    
    # Ограничиваем до 5 одновременных запросов
    semaphore = asyncio.Semaphore(5)
    
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_url(session, url, semaphore) for url in urls]
        results = await asyncio.gather(*tasks)
        print(f"Загружено {len(results)} страниц")

asyncio.run(main())

Тайм-ауты и обработка ошибок

import asyncio
import aiohttp

async def fetch_with_timeout(session, url, timeout=5):
    try:
        # Устанавливаем тайм-аут для запроса
        async with session.get(url, timeout=timeout) as response:
            return await response.text()
    except asyncio.TimeoutError:
        print(f"Тайм-аут при запросе к {url}")
        return None
    except aiohttp.ClientError as e:
        print(f"Ошибка клиента при запросе к {url}: {e}")
        return None
    except Exception as e:
        print(f"Неизвестная ошибка при запросе к {url}: {e}")
        return None

async def main():
    timeout = aiohttp.ClientTimeout(total=3)
    async with aiohttp.ClientSession(timeout=timeout) as session:
        urls = [
            "https://jsonplaceholder.typicode.com/posts/1",
            "https://несуществующий-сайт.ру",
            "https://jsonplaceholder.typicode.com/posts/2",
        ]
        
        tasks = [fetch_with_timeout(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
        
        for i, result in enumerate(results):
            print(f"Результат {i}: {result is not None}")

asyncio.run(main())

import asyncio
import aiofiles

async def write_file(filename, content):
    """Асинхронная запись в файл"""
    async with aiofiles.open(filename, 'w', encoding='utf-8') as f:
        await f.write(content)
    print(f"Файл {filename} записан")

async def read_file(filename):
    """Асинхронное чтение файла"""
    async with aiofiles.open(filename, 'r', encoding='utf-8') as f:
        content = await f.read()
    return content

async def main():
    # Запись нескольких файлов параллельно
    tasks = [
        write_file('file1.txt', 'Содержимое файла 1'),
        write_file('file2.txt', 'Содержимое файла 2'),
        write_file('file3.txt', 'Содержимое файла 3'),
    ]
    await asyncio.gather(*tasks)
    
    # Чтение файлов параллельно
    read_tasks = [
        read_file('file1.txt'),
        read_file('file2.txt'),
        read_file('file3.txt'),
    ]
    contents = await asyncio.gather(*read_tasks)
    
    for i, content in enumerate(contents, 1):
        print(f"Файл {i}: {content}")

asyncio.run(main())

Копирование больших файлов

import asyncio
import aiofiles

async def copy_file_chunked(src, dst, chunk_size=1024*1024):
    """Копирование файла чанками (для больших файлов)"""
    async with aiofiles.open(src, 'rb') as src_file:
        async with aiofiles.open(dst, 'wb') as dst_file:
            while True:
                chunk = await src_file.read(chunk_size)
                if not chunk:
                    break
                await dst_file.write(chunk)
    
    print(f"Файл {src} скопирован в {dst}")

async def main():
    # Копируем несколько файлов параллельно
    files_to_copy = [
        ('big_file1.bin', 'copy1.bin'),
        ('big_file2.bin', 'copy2.bin'),
        ('big_file3.bin', 'copy3.bin'),
    ]
    
    tasks = [copy_file_chunked(src, dst) for src, dst in files_to_copy]
    await asyncio.gather(*tasks)

Урок 5: Синхронизация и очереди

Asyncio.Queue

Очереди позволяют организовать взаимодействие между задачами (производитель-потребитель).

import asyncio
import random

async def producer(queue, name):
    """Производитель генерирует данные"""
    for i in range(5):
        item = f"Товар {i} от {name}"
        await asyncio.sleep(random.random())  # Случайная задержка
        await queue.put(item)
        print(f"[{name}] Произвел: {item}")
    
    # Сигнал о завершении
    await queue.put(None)

async def consumer(queue, name):
    """Потребитель обрабатывает данные"""
    while True:
        item = await queue.get()
        if item is None:
            # Возвращаем None обратно для других потребителей
            await queue.put(None)
            break
        
        await asyncio.sleep(random.random() * 2)  # Обработка занимает время
        print(f"[{name}] Обработал: {item}")
        queue.task_done()

async def main():
    queue = asyncio.Queue(maxsize=3)  # Очередь максимум на 3 элемента
    
    # Запускаем производителей и потребителей
    producers = [
        asyncio.create_task(producer(queue, f"Производитель_{i}"))
        for i in range(2)
    ]
    
    consumers = [
        asyncio.create_task(consumer(queue, f"Потребитель_{i}"))
        for i in range(3)
    ]
    
    # Ждем завершения производителей
    await asyncio.gather(*producers)
    
    # Ждем завершения обработки всех элементов
    await queue.join()
    
    # Отменяем потребителей
    for c in consumers:
        c.cancel()

asyncio.run(main())

# Семафор - ограничивает количество одновременных доступов
async def worker(semaphore, name):

    await asyncio.sleep(2)
    print("Событие происходит!")

async def event_example():
    event = asyncio.Event()

        waiter(event, f"Waiter_{i}") for i in range(3)
    ]
    tasks.append(setter(event))
    await asyncio.gather(*tasks)

counter = 0
lock = asyncio.Lock()

async def increment():
    global counter
    async with lock:
        temp = counter
        await asyncio.sleep(0.1)  # Имитация работы
        counter = temp + 1
        print(f"Счетчик: {counter}")

async def lock_example():
    tasks = [increment() for _ in range(10)]
    await asyncio.gather(*tasks)
    print(f"Итоговый счетчик: {counter}")

Урок 6: Многопоточность и многопроцессность

Сравнение подходов

Подход Когда использовать Плюсы Минусы
asyncio I/O-bound задачи (сеть, файлы) Очень легковесный, тысячи задач Специфичный синтаксис, не для CPU
threading I/O-bound, блокирующие операции Привычный, параллельный I/O GIL, сложнее синхронизация
multiprocessing CPU-bound задачи Обходит GIL, настоящий параллелизм Тяжеловесный, сложнее обмен данными

Многопоточность (threading)

import threading
import time

def worker(name, seconds):
    print(f"Поток {name} начал работу")
    time.sleep(seconds)  # Блокирующая операция
    print(f"Поток {name} завершил работу")
    return f"Результат {name}"

# Создание и запуск потоков
threads = []
for i in range(5):
    t = threading.Thread(target=worker, args=(f"Thread-{i}", i))
    t.start()
    threads.append(t)

# Ожидание завершения
for t in threads:
    t.join()

print("Все потоки завершены")

# С передачей результатов (через очередь)
import queue

def worker_with_queue(name, q):
    result = f"Результат от {name}"
    time.sleep(1)
    q.put(result)

q = queue.Queue()
threads = []
for i in range(3):
    t = threading.Thread(target=worker_with_queue, args=(f"Thread-{i}", q))
    t.start()
    threads.append(t)

for t in threads:
    t.join()

while not q.empty():
    print(q.get())

Многопроцессность (multiprocessing)

import multiprocessing
import time
import math

def cpu_intensive(n):
    """CPU-интенсивная задача"""
    print(f"Процесс {multiprocessing.current_process().name} начал вычисления")
    result = sum(math.sqrt(i) for i in range(n))
    print(f"Процесс завершил вычисления")
    return result

if __name__ == "__main__":
    # Создание пула процессов
    with multiprocessing.Pool(processes=4) as pool:
        numbers = [10_000_000, 20_000_000, 30_000_000, 40_000_000]
        
        # Параллельное выполнение
        start = time.time()
        results = pool.map(cpu_intensive, numbers)
        print(f"Время с multiprocessing: {time.time() - start:.2f} сек")
        print(f"Результаты: {results}")
    
    # Для сравнения - последовательное выполнение
    start = time.time()
    sequential_results = [cpu_intensive(n) for n in numbers]
    print(f"Время последовательно: {time.time() - start:.2f} сек")

# Обмен данными между процессами
from multiprocessing import Process, Queue

def producer(q):
    for i in range(5):
        q.put(f"Данные {i}")
    q.put(None)  # Сигнал завершения

def consumer(q):
    while True:
        data = q.get()
        if data is None:
            break
        print(f"Потребитель получил: {data}")

if __name__ == "__main__":
    q = Queue()
    p1 = Process(target=producer, args=(q,))
    p2 = Process(target=consumer, args=(q,))
    
    p1.start()
    p2.start()
    
    p1.join()

Concurrent.futures (высокоуровневый API)

import concurrent.futures
import time
import math

def io_bound_task(seconds):
    """Имитация I/O операции"""
    time.sleep(seconds)
    return f"Завершено за {seconds} сек"

def cpu_bound_task(n):
    """Имитация CPU операции"""
    return sum(i * i for i in range(n))

# ThreadPoolExecutor для I/O-bound задач
with concurrent.futures.ThreadPoolExecutor(max_workers=3) as executor:
    futures = [executor.submit(io_bound_task, i) for i in [1, 2, 3]]
    
    for future in concurrent.futures.as_completed(futures):
        print(future.result())

# ProcessPoolExecutor для CPU-bound задач
with concurrent.futures.ProcessPoolExecutor(max_workers=4) as executor:
    numbers = [10_000_000, 20_000_000, 30_000_000]
    results = list(executor.map(cpu_bound_task, numbers))
    print(f"Результаты: {results}")

# Map для удобства
with concurrent.futures.ThreadPoolExecutor() as executor:
    urls = ['url1', 'url2', 'url3']
    results = executor.map(io_bound_task, [1, 2, 1])
    for result in results:
        print(result)

Проект: Парсер новостей

Создадим асинхронный парсер, который собирает заголовки новостей с нескольких сайтов.

import asyncio
import aiohttp
from bs4 import BeautifulSoup
import time

class AsyncNewsParser:
    def __init__(self, max_concurrent=5):
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.results = []
    
    async def fetch_page(self, session, url):
        """Асинхронная загрузка страницы"""
        async with self.semaphore:
            try:
                print(f"Загрузка: {url}")
                async with session.get(url, timeout=10) as response:
                    html = await response.text()
                    print(f"✓ Загружено: {url} ({len(html)} символов)")
                    return url, html
            except Exception as e:
                print(f"✗ Ошибка загрузки {url}: {e}")
                return url, None
    
    async def parse_page(self, url, html):
        """Парсинг страницы (может быть CPU-bound, поэтому в отдельном потоке)"""
        if not html:
            return None
        
        # Используем run_in_executor для CPU-bound парсинга
        loop = asyncio.get_event_loop()
        soup = await loop.run_in_executor(None, BeautifulSoup, html, 'html.parser')
        
        # Извлекаем заголовки (зависит от структуры сайта)
        titles = []
        for h2 in soup.find_all(['h2', 'h3'])[:5]:
            text = h2.get_text().strip()
            if text and len(text) > 10:
                titles.append(text)
        
        return {
            'url': url,
            'titles': titles,
            'count': len(titles)
        }
    
    async def process_site(self, session, url):
        """Обработка одного сайта"""
        url, html = await self.fetch_page(session, url)
        result = await self.parse_page(url, html)
        if result:
            self.results.append(result)
            print(f"  Найдено {result['count']} заголовков на {url}")
    
    async def run(self, urls):
        """Запуск парсера"""
        async with aiohttp.ClientSession() as session:
            tasks = [self.process_site(session, url) for url in urls]
            await asyncio.gather(*tasks)
        return self.results

async def main():
    # Список сайтов для парсинга
    urls = [
        'https://news.ycombinator.com',
        'https://www.python.org/blogs/',
        'https://realpython.com/',
        'https://habr.com/ru/hub/python/',
        'https://dev.to/t/python',
        'https://www.reddit.com/r/Python/',
    ]
    
    parser = AsyncNewsParser(max_concurrent=3)
    
    start = time.time()
    results = await parser.run(urls)
    elapsed = time.time() - start
    
    print("\n" + "="*50)
    print(f"Парсинг завершен за {elapsed:.2f} секунд")
    print(f"Обработано сайтов: {len(results)}")
    print("="*50)
    
    for result in results:
        print(f"\n{result['url']}:")
        for i, title in enumerate(result['titles'], 1):
            print(f"  {i}. {title[:70]}...")

if __name__ == "__main__":
    asyncio.run(main())

Этот проект демонстрирует:

  • Асинхронные HTTP-запросы (aiohttp)
  • Ограничение конкурентности (семафор)
  • Обработку ошибок и таймаутов
  • Комбинацию asyncio с CPU-bound кодом (run_in_executor)
  • Сбор и агрегацию результатов
