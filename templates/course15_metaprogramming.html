Урок 1: Что такое декораторы?

Декораторы — это обёртки для функций

Декоратор — это функция, которая принимает другую функцию и расширяет её поведение без изменения исходного кода. Это мощный инструмент Python, который позволяет добавлять функциональность (логирование, проверку прав, кэширование и т.д.) к существующим функциям.

# Простой декоратор без синтаксического сахара
def decorator(func):
    def wrapper():
        print("Что-то происходит ДО вызова функции")
        func()
        print("Что-то происходит ПОСЛЕ вызова функции")
    return wrapper

def say_hello():
    print("Привет, мир!")

# Применяем декоратор
say_hello = decorator(say_hello)
say_hello()

# Результат:
# Что-то происходит ДО вызова функции
# Привет, мир!
# Что-то происходит ПОСЛЕ вызова функции

Синтаксис @decorator

Python предоставляет удобный синтаксис с символом @ для применения декораторов.

# То же самое, но с синтаксисом @
def decorator(func):
    def wrapper():
        print("До вызова")
        func()
        print("После вызова")
    return wrapper

@decorator
def say_hello():
    print("Привет, мир!")

say_hello()

Важно! Декоратор выполняется в момент определения функции, а не в момент её вызова. Декоратор возвращает новую функцию, которая и будет вызвана вместо оригинальной.

Урок 2: Декораторы с аргументами

Декораторы с параметрами

Иногда нужно передавать аргументы в сам декоратор. Для этого используется трёхуровневая вложенность.

def repeat(times):
    """Декоратор, который повторяет функцию указанное количество раз"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                result = func(*args, **kwargs)
                results.append(result)
            return results
        return wrapper
    return decorator

@repeat(3)
def greet(name):
    return f"Привет, {name}!"

print(greet("Анна"))
# ['Привет, Анна!', 'Привет, Анна!', 'Привет, Анна!']

# Декоратор с проверкой типов
def type_check(*types):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for arg, expected_type in zip(args, types):
                if not isinstance(arg, expected_type):
                    raise TypeError(f"Аргумент {arg} должен быть {expected_type}")
            return func(*args, **kwargs)
        return wrapper
    return decorator

@type_check(int, int)
def add(a, b):
    return a + b

print(add(5, 3))   # 8
# print(add("5", 3))  # TypeError!

Декоратор для измерения времени выполнения

import time

def timer(func):
    """Декоратор, измеряющий время выполнения функции"""
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} выполнилась за {end - start:.4f} сек")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "Готово"

slow_function()  # slow_function выполнилась за 1.0012 сек

Урок 3: Декораторы для классов

Декораторы классов

Декораторы можно применять не только к функциям, но и к классам. Они позволяют изменять поведение классов, добавлять методы или атрибуты.

def add_repr(cls):
    """Декоратор, добавляющий метод __repr__ в класс"""
    def __repr__(self):
        attrs = []
        for key, value in self.__dict__.items():
            attrs.append(f"{key}={value!r}")
        return f"{cls.__name__}({', '.join(attrs)})"
    
    cls.__repr__ = __repr__
    return cls

@add_repr
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

p = Person("Анна", 25)
print(p)  # Person(name='Анна', age=25)

# Декоратор для Singleton (одиночка)
def singleton(cls):
    instances = {}
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return get_instance

@singleton
class DatabaseConnection:
    def __init__(self):
        self.connected = True

db1 = DatabaseConnection()
db2 = DatabaseConnection()
print(db1 is db2)  # True

Встроенные декораторы классов

from dataclasses import dataclass

# @dataclass - автоматически добавляет __init__, __repr__, __eq__
@dataclass
class Point:
    x: int
    y: int

p1 = Point(1, 2)
p2 = Point(1, 2)
print(p1 == p2)  # True
print(p1)        # Point(x=1, y=2)

# @property - делает метод атрибутом
class Circle:
    def __init__(self, radius):
        self._radius = radius
    
    @property
    def radius(self):
        return self._radius
    
    @radius.setter
    def radius(self, value):
        if value < 0:
            raise ValueError("Радиус не может быть отрицательным")
        self._radius = value
    
    @property
    def area(self):
        return 3.14159 * self.radius ** 2

c = Circle(5)
c.radius = 10  # сеттер
print(c.area)  # геттер - 314.159

Урок 4: Практические декораторы

Декоратор для проверки прав доступа

def require_permission(permission):
    """Декоратор для проверки прав доступа"""
    def decorator(func):
        def wrapper(user, *args, **kwargs):
            if not user.get(permission, False):
                raise PermissionError(f"Нет права '{permission}'")
            return func(user, *args, **kwargs)
        return wrapper
    return decorator

@require_permission("admin")
def delete_user(user, user_id):
    return f"Пользователь {user_id} удалён"

# Проверка
admin = {"name": "Admin", "admin": True}
user = {"name": "User", "admin": False}

print(delete_user(admin, 123))  # Пользователь 123 удалён
# print(delete_user(user, 123))  # PermissionError!

Декоратор кэширования (LRU Cache)

from functools import lru_cache
import time

# Встроенный декоратор для кэширования
@lru_cache(maxsize=128)
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

start = time.time()
print(fibonacci(35))
print(f"Время: {time.time() - start:.4f} сек")  # очень быстро!

# Реализация своего декоратора кэширования
def cache(func):
    _cache = {}
    def wrapper(*args):
        if args not in _cache:
            _cache[args] = func(*args)
        return _cache[args]
    return wrapper

@cache
def factorial(n):
    if n <= 1:
        return 1
    return n * factorial(n-1)

print(factorial(10))  # 3628800
print(factorial(10))  # результат из кэша

Декоратор повторных попыток (Retry)

import time
import random

def retry(max_attempts=3, delay=1):
    """Декоратор для повторных попыток при ошибке"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    print(f"Попытка {attempt} не удалась: {e}")
                    time.sleep(delay)
            return None
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unstable_function():
    if random.random() < 0.7:
        raise ConnectionError("Соединение потеряно")
    return "Успешно!"

print(unstable_function())

Декоратор логирования

import functools
import time

def log(level="INFO"):
    """Декоратор для логирования вызовов функций"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            args_str = ', '.join(map(str, args))
            kwargs_str = ', '.join(f"{k}={v}" for k, v in kwargs.items())
            all_args = ', '.join(filter(None, [args_str, kwargs_str]))
            
            print(f"[{level}] Вызов {func.__name__}({all_args})")
            start = time.time()
            result = func(*args, **kwargs)
            elapsed = time.time() - start
            print(f"[{level}] {func.__name__} -> {result} ({elapsed:.4f} сек)")
            return result
        return wrapper
    return decorator

@log(level="DEBUG")
def multiply(a, b):
    return a * b

@log(level="ERROR")
def divide(a, b):
    return a / b

multiply(5, 3)
divide(10, 2)

Урок 5: Введение в метаклассы

Что такое метакласс?

Метакласс — это класс, который создаёт классы. Если класс — это "чертёж" для объектов, то метакласс — это "чертёж" для классов. По умолчанию все классы в Python создаются метаклассом type.

# Обычный класс
class MyClass:
    pass

# Эквивалентное создание через type
MyClass = type('MyClass', (), {})

# type(name, bases, dict)
# name: имя класса
# bases: кортеж родительских классов
# dict: словарь атрибутов и методов

# Создание класса с атрибутами и методами
def say_hello(self):
    return "Привет!"

Person = type('Person', (), {
    'name': 'Аноним',
    'age': 0,
    'greet': say_hello
})

p = Person()
print(p.name)      # Аноним
print(p.greet())   # Привет!

Создание своего метакласса

class UpperCaseMeta(type):
    """Метакласс, который преобразует все атрибуты класса в верхний регистр"""
    def __new__(cls, name, bases, attrs):
        # Преобразуем имена методов и атрибутов
        upper_attrs = {}
        for attr_name, attr_value in attrs.items():
            if not attr_name.startswith('__'):
                upper_attrs[attr_name.upper()] = attr_value
            else:
                upper_attrs[attr_name] = attr_value
        return super().__new__(cls, name, bases, upper_attrs)

class MyClass(metaclass=UpperCaseMeta):
    my_attribute = "hello"
    my_method = lambda self: "world"

obj = MyClass()
print(obj.MY_ATTRIBUTE)  # hello
print(obj.MY_METHOD())   # world

Метакласс для проверки аннотаций типов

class TypeCheckMeta(type):
    """Метакласс, проверяющий типы при инициализации"""
    def __call__(cls, *args, **kwargs):
        # Получаем аннотации типов из класса
        annotations = getattr(cls, '__annotations__', {})
        
        # Проверяем позиционные аргументы
        for i, (name, expected_type) in enumerate(annotations.items()):
            if i < len(args):
                arg = args[i]
                if not isinstance(arg, expected_type):
                    raise TypeError(f"{name} должен быть {expected_type}, получен {type(arg)}")
        
        # Проверяем именованные аргументы
        for name, value in kwargs.items():
            if name in annotations:
                expected_type = annotations[name]
                if not isinstance(value, expected_type):
                    raise TypeError(f"{name} должен быть {expected_type}, получен {type(value)}")
        
        return super().__call__(*args, **kwargs)

class Person(metaclass=TypeCheckMeta):
    name: str
    age: int
    
    def __init__(self, name, age):
        self.name = name
        self.age = age

p1 = Person("Анна", 25)   # OK
# p2 = Person(123, "25")   # TypeError!

Важно! Метаклассы — мощный, но сложный инструмент. Используйте их только когда действительно нужно. В большинстве случаев декораторов и наследования достаточно.

Урок 6: Магические (dunder) методы

Основные магические методы

Магические методы (dunder methods — от double underscore) позволяют определять поведение объектов для встроенных операций.

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __repr__(self):
        return f"Vector({self.x}, {self.y})"
    
    def __add__(self, other):
        """Оператор +"""
        return Vector(self.x + other.x, self.y + other.y)
    
    def __sub__(self, other):
        """Оператор -"""
        return Vector(self.x - other.x, self.y - other.y)
    
    def __mul__(self, scalar):
        """Оператор *"""
        return Vector(self.x * scalar, self.y * scalar)
    
    def __eq__(self, other):
        """Оператор =="""
        return self.x == other.x and self.y == other.y
    
    def __len__(self):
        """Длина вектора (расстояние до начала координат)"""
        return int((self.x**2 + self.y**2) ** 0.5)
    
    def __getitem__(self, key):
        """Доступ по индексу"""
        if key == 0:
            return self.x
        elif key == 1:
            return self.y
        raise IndexError("Индекс должен быть 0 или 1")
    
    def __call__(self, factor):
        """Вызов объекта как функции"""
        return Vector(self.x * factor, self.y * factor)

v1 = Vector(3, 4)
v2 = Vector(1, 2)

print(v1 + v2)     # Vector(4, 6)
print(v1 - v2)     # Vector(2, 2)
print(v1 * 2)      # Vector(6, 8)
print(v1 == v2)    # False
print(len(v1))     # 5
print(v1[0])       # 3
print(v1(5))       # Vector(15, 20)

Магические методы для контейнеров

class SparseArray:
    """Разрежённый массив (только непустые элементы)"""
    def __init__(self):
        self._data = {}
    
    def __setitem__(self, key, value):
        """Установка значения по индексу"""
        self._data[key] = value
    
    def __getitem__(self, key):
        """Получение значения по индексу"""
        return self._data.get(key, 0)
    
    def __contains__(self, item):
        """Оператор in"""
        return item in self._data
    
    def __iter__(self):
        """Итерация по ключам"""
        return iter(self._data.keys())
    
    def __len__(self):
        return len(self._data)

arr = SparseArray()
arr[10] = 100
arr[1000] = 500

print(arr[10])     # 100
print(arr[5])      # 0
print(10 in arr)   # True
print(len(arr))    # 2
for i in arr:
    print(i)       # 10, 1000

Магические методы для сравнения

from functools import total_ordering

@total_ordering  # Автоматически добавляет остальные методы сравнения
class Student:
    def __init__(self, name, grade):
        self.name = name
        self.grade = grade
    
    def __eq__(self, other):
        return self.grade == other.grade
    
    def __lt__(self, other):
        return self.grade < other.grade
    
    def __repr__(self):
        return f"{self.name}: {self.grade}"

students = [
    Student("Анна", 85),
    Student("Иван", 92),
    Student("Мария", 78)
]

print(sorted(students))  # [Мария: 78, Анна: 85, Иван: 92]
print(max(students))     # Иван: 92

Урок 8: Практический проект — Фреймворк маршрутизации

Создаём простой веб-фреймворк

Используя декораторы и метаклассы, создадим мини-фреймворк для маршрутизации HTTP-запросов, похожий на Flask.

import inspect
import re

class Route:
    """Декоратор для регистрации маршрута"""
    def __init__(self, path, methods=None):
        self.path = path
        self.methods = methods or ['GET']
    
    def __call__(self, handler):
        handler._route = self.path
        handler._methods = self.methods
        return handler


class AppMeta(type):
    """Метакласс для сбора маршрутов"""
    def __new__(cls, name, bases, attrs):
        routes = {}
        
        for attr_name, attr_value in attrs.items():
            if callable(attr_value) and hasattr(attr_value, '_route'):
                routes[attr_value._route] = {
                    'handler': attr_value,
                    'methods': getattr(attr_value, '_methods', ['GET'])
                }
        
        attrs['_routes'] = routes
        return super().__new__(cls, name, bases, attrs)


class Application(metaclass=AppMeta):
    """Основной класс приложения"""
    _routes = {}
    
    def __init__(self):
        self.routes = self.__class__._routes
    
    def dispatch(self, path, method='GET'):
        """Находит и вызывает обработчик для пути"""
        for route_path, route_info in self.routes.items():
            if route_path == path and method in route_info['methods']:
                return route_info['handler'](self)
        
        return "404 Not Found"
    
    def run(self):
        """Запуск сервера (эмуляция)"""
        print("Сервер запущен на http://localhost:8000")
        print("Доступные маршруты:")
        for path, info in self.routes.items():
            print(f"  {path} ({', '.join(info['methods'])})")
        
        # Эмуляция запросов
        print("\n" + "="*40)
        print("Тестирование маршрутов:")
        print(self.dispatch('/'))
        print(self.dispatch('/about'))
        print(self.dispatch('/user/42'))


# Использование фреймворка
class MyApp(Application):
    
    @Route('/')
    def index(self):
        return "Добро пожаловать на главную страницу!"
    
    @Route('/about')
    def about(self):
        return "О нас: Это учебный проект"
    
    @Route('/user/')
    def user_profile(self, id=None):
        return f"Профиль пользователя #{id}"


# Запуск
app = MyApp()
app.run()

Этот проект демонстрирует:

  • Создание декораторов с параметрами (@Route)
  • Использование метаклассов для сбора маршрутов
  • Рефлексию (проверку атрибутов у функций)
  • Реализацию простого диспетчера запросов
