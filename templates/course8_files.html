Урок 1: Открытие и закрытие файлов

Функция open()

Для работы с файлами в Python используется функция open(), которая возвращает файловый объект.

# Базовый синтаксис
file = open('filename.txt', 'mode')
# ... работа с файлом
file.close()  # Важно закрывать файл!

Режимы открытия файлов:

Режим Описание
'r' Чтение (по умолчанию). Файл должен существовать.
'w' Запись. Создает новый файл или перезаписывает существующий.
'a' Дозапись в конец файла. Создает файл, если его нет.
'x' Создание. Ошибка, если файл уже существует.
'b' Бинарный режим (добавляется к другим, например 'rb')
't' Текстовый режим (по умолчанию, например 'rt')
'+' Чтение и запись (например 'r+') 
# Примеры открытия файлов
file1 = open('data.txt', 'r')      # Чтение текстового файла
file2 = open('output.txt', 'w')    # Запись (перезапись)
file3 = open('log.txt', 'a')       # Дозапись в конец
file4 = open('image.jpg', 'rb')    # Чтение бинарного файла

Контекстный менеджер with

Лучший способ работы с файлами — использовать конструкцию with. Она автоматически закрывает файл, даже если произошла ошибка.

# Рекомендуемый способ
with open('filename.txt', 'r') as file:
    content = file.read()
    print(content)
# Файл автоматически закрывается после выхода из блока

# Можно открывать несколько файлов
with open('input.txt', 'r') as infile, open('output.txt', 'w') as outfile:
    data = infile.read()
    outfile.write(data.upper())

Важно! Всегда закрывайте файлы! При использовании with это происходит автоматически. Если забыть close(), файл может остаться заблокированным, а данные не сохранятся.

Урок 2: Чтение из файлов

Методы чтения

1. read() — читает весь файл целиком

with open('story.txt', 'r', encoding='utf-8') as file:
    content = file.read()
    print(content)  # Весь текст файла

2. readline() — читает одну строку

with open('story.txt', 'r', encoding='utf-8') as file:
    line1 = file.readline()  # Первая строка
    line2 = file.readline()  # Вторая строка
    print(line1, line2)

3. readlines() — читает все строки в список

with open('story.txt', 'r', encoding='utf-8') as file:
    lines = file.readlines()
    for i, line in enumerate(lines, 1):
        print(f"Строка {i}: {line.rstrip()}")

4. Итерация по файлу — эффективно для больших файлов

with open('large_file.txt', 'r', encoding='utf-8') as file:
    for line in file:  # Читает построчно, не загружая всё в память
        print(line.strip())

Параметр encoding

Важно указывать кодировку, особенно для русского текста. По умолчанию используется кодировка платформы.

# Чтение русского текста
with open('russian.txt', 'r', encoding='utf-8') as file:
    text = file.read()
    
# Другие популярные кодировки
# encoding='cp1251'  # Windows-1251

Урок 3: Запись в файлы

Методы записи

1. write() — записывает строку в файл

with open('output.txt', 'w', encoding='utf-8') as file:
    file.write("Привет, мир!\n")
    file.write("Это вторая строка.\n")
# Файл будет создан или перезаписан

2. writelines() — записывает список строк

lines = [
    "Первая строка\n",
    "Вторая строка\n",
    "Третья строка\n"
]

with open('output.txt', 'w', encoding='utf-8') as file:
    file.writelines(lines)
# writelines НЕ добавляет переносы строк автоматически!

3. Режим 'a' (append) — дозапись в конец файла

with open('log.txt', 'a', encoding='utf-8') as file:
    file.write("Новая запись в лог\n")
    file.write("Еще одна запись\n")
# Существующий файл не перезаписывается

Форматированный вывод

# Использование f-строк
name = "Анна"
age = 25
with open('users.txt', 'a', encoding='utf-8') as file:
    file.write(f"Имя: {name}, Возраст: {age}\n")

# Использование форматирования
data = [1, 2, 3, 4, 5]
with open('numbers.txt', 'w') as file:
    for num in data:
        file.write(f"{num}\n")

Урок 4: Работа с CSV

Модуль csv

CSV (Comma-Separated Values) — популярный формат для хранения табличных данных.

import csv

# Чтение CSV файла
with open('users.csv', 'r', encoding='utf-8') as file:
    reader = csv.reader(file)
    for row in reader:
        print(row)  # Каждая строка - список

# Чтение с заголовками (Dictionary)
with open('users.csv', 'r', encoding='utf-8') as file:
    reader = csv.DictReader(file)
    for row in reader:
        print(row['name'], row['age'])
# Запись в CSV
data = [
    ['name', 'age', 'city'],
    ['Анна', 25, 'Москва'],
    ['Иван', 30, 'СПб'],
    ['Мария', 28, 'Казань']
]

with open('users.csv', 'w', newline='', encoding='utf-8') as file:
    writer = csv.writer(file)
    writer.writerows(data)

# Запись словарей
users = [
    {'name': 'Анна', 'age': 25, 'city': 'Москва'},
    {'name': 'Иван', 'age': 30, 'city': 'СПб'}
]

with open('users.csv', 'w', newline='', encoding='utf-8') as file:
    writer = csv.DictWriter(file, fieldnames=['name', 'age', 'city'])
    writer.writeheader()  # Записываем заголовки
    writer.writerows(users)

Совет: Параметр newline='' в Windows предотвращает добавление лишних пустых строк.

Урок 5: Работа с JSON

Модуль json

JSON (JavaScript Object Notation) — удобный формат для обмена данными.

import json

# Python словарь
data = {
    'name': 'Анна',
    'age': 25,
    'city': 'Москва',
    'hobbies': ['чтение', 'плавание'],
    'married': False
}

# Преобразование в JSON строку
json_string = json.dumps(data, ensure_ascii=False, indent=2)
print(json_string)

# Запись JSON в файл
with open('data.json', 'w', encoding='utf-8') as file:
    json.dump(data, file, ensure_ascii=False, indent=2)

# Чтение JSON из файла
with open('data.json', 'r', encoding='utf-8') as file:
    loaded_data = json.load(file)
    print(loaded_data['name'])
    print(loaded_data['hobbies'])

dict object

Урок 6: Работа с файловой системой

Модули os и shutil

import os
import shutil

# Текущая директория
print(os.getcwd())

# Список файлов в директории
files = os.listdir('.')
print(files)

# Создание директории
os.mkdir('new_folder')  # одна папка
os.makedirs('folder/subfolder')  # создает все промежуточные

# Проверка существования
if os.path.exists('file.txt'):
    print("Файл существует")

# Переименование/перемещение
os.rename('old.txt', 'new.txt')

# Удаление файла
os.remove('file.txt')

# Удаление директории
os.rmdir('empty_folder')  # только пустую
shutil.rmtree('folder')    # удаляет всё содержимое

# Копирование файлов
shutil.copy('source.txt', 'destination.txt')
shutil.copytree('source_dir', 'dest_dir')  # копирует папку

Работа с путями

from pathlib import Path

# Современный способ работы с путями
path = Path('folder/subfolder/file.txt')

print(path.name)        # file.txt
print(path.stem)        # file (без расширения)
print(path.suffix)      # .txt
print(path.parent)      # folder/subfolder
print(path.parents[0])  # folder/subfolder

# Создание путей
new_path = Path('data') / 'images' / 'photo.jpg'

# Проверка
if path.exists():
    if path.is_file():
        print("Это файл")
    if path.is_dir():
        print("Это папка")

# Обход директории
for file_path in Path('.').glob('*.txt'):
    print(file_path)

# Рекурсивный обход
for file_path in Path('.').rglob('*.py'):
    print(file_path)
