Урок 1: Модули и импорт

Что такое модуль?

Модуль — это просто файл с кодом на Python с расширением .py. Модули позволяют организовывать код, разделять его на логические части и переиспользовать в разных программах.

Способы импорта:

# 1. Импорт всего модуля
import math
print(math.sqrt(25))
print(math.pi)

# 2. Импорт с псевдонимом
import datetime as dt
today = dt.date.today()
print(today)

# 3. Импорт конкретных функций/переменных
from random import randint, choice
print(randint(1, 10))
print(choice(['камень', 'ножницы', 'бумага']))

# 4. Импорт всех имен (не рекомендуется!)
from os import *  # Может привести к конфликтам имен

Пути поиска модулей

Python ищет модули в следующих местах (в порядке приоритета):

  1. Текущая директория
  2. PYTHONPATH (переменная окружения)
  3. Стандартные пути установки Python
  4. Пути в site-packages (для сторонних библиотек)
import sys

# Список путей поиска модулей
for path in sys.path:
    print(path)

Урок 2: Создание своих модулей

Создание простого модуля

Создадим файл my_utils.py:

# Файл: my_utils.py

def greet(name):
    """Приветствие"""
    return f"Привет, {name}!"

def add(a, b):
    """Сложение двух чисел"""
    return a + b

def is_even(num):
    """Проверка на четность"""
    return num % 2 == 0

PI = 3.14159
AUTHOR = "PyMaster"

Теперь можем использовать этот модуль в другой программе:

# Файл: main.py
import my_utils

print(my_utils.greet("Анна"))
print(my_utils.add(10, 5))
print(my_utils.is_even(7))
print(my_utils.PI)

Защита от выполнения при импорте

Иногда нужно, чтобы какой-то код выполнялся только при прямом запуске файла, но не при импорте. Для этого используется if __name__ == "__main__":

# Файл: calculator.py

def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

# Этот код выполнится ТОЛЬКО при запуске файла напрямую
if __name__ == "__main__":
    print("Тестирование модуля:")
    print(f"5 + 3 = {add(5, 3)}")
    print(f"10 - 4 = {subtract(10, 4)}")

При импорте этого модуля тестовый код не выполнится:

import calculator  # Тестовый код не запустится

result = calculator.add(100, 50)
print(result)  # 150

Урок 3: Пакеты

Структура пакета

Пакет — это папка, содержащая несколько модулей и специальный файл __init__.py. Пакеты позволяют создавать иерархическую структуру проекта.

Пример структуры пакета:

my_package/
│
├── __init__.py
├── math_ops/
│   ├── __init__.py
│   ├── basic.py
│   └── advanced.py
├── string_ops/
│   ├── __init__.py
│   ├── formatting.py
│   └── validation.py
└── utils.py

Содержимое модулей:

# math_ops/basic.py
def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

# math_ops/advanced.py
def power(a, b):
    return a ** b

def factorial(n):
    if n <= 1:
        return 1
    return n * factorial(n - 1)

# __init__.py может быть пустым или содержать код инициализации

Использование пакета:

# Импорт из пакета
from my_package.math_ops.basic import add, subtract
from my_package.math_ops.advanced import power, factorial

print(add(10, 5))        # 15
print(factorial(5))      # 120

Модуль math

import math

# Константы
print(math.pi)       # 3.141592653589793
print(math.e)        # 2.718281828459045

# Основные функции
print(math.sqrt(25))      # 5.0 (квадратный корень)
print(math.pow(2, 10))    # 1024.0 (возведение в степень)
print(math.factorial(5))  # 120 (факториал)

# Тригонометрия
print(math.sin(math.pi/2))    # 1.0
print(math.cos(0))            # 1.0

# Округление
print(math.ceil(4.2))    # 5 (вверх)
print(math.floor(4.9))   # 4 (вниз)

Модуль datetime

import datetime

now = datetime.datetime.now()
print(now)

today = datetime.date.today()

Урок 5: pip и сторонние библиотеки

Менеджер пакетов pip

pip — стандартный менеджер пакетов Python для установки сторонних библиотек из PyPI (Python Package Index).

Основные команды:

# Установка пакета
pip install requests

# Установка конкретной версии
pip install requests==2.28.1

# Удаление пакета
pip uninstall requests

# Список установленных пакетов
pip list

# Информация о пакете
pip show requests

# Сохранение списка зависимостей
pip freeze > requirements.txt

# Установка из requirements.txt
pip install -r requirements.txt

requests — работа с HTTP

import requests

# GET запрос
response = requests.get('https://api.github.com')
print(response.status_code)  # 200
print(response.json())       # Ответ в формате JSON

# GET с параметрами
params = {'q': 'python', 'page': 1}
response = requests.get('https://api.github.com/search/repositories', params=params)

# POST запрос
data = {'title': 'Новая задача', 'completed': False}
response = requests.post('https://jsonplaceholder.typicode.com/todos', json=data)

# Обработка ошибок
try:
    response = requests.get('https://несуществующий-сайт.ру', timeout=5)
    response.raise_for_status()  # Выбросит исключение при ошибке HTTP
except requests.exceptions.RequestException as e:
    print(f"Ошибка запроса: {e}")

Урок 7: Практика с модулями

Практическое задание

Создайте свой пакет для математических операций и программу для его тестирования.

Структура:

math_package/
│
├── __init__.py
├── basic.py      # сложение, вычитание, умножение, деление
├── advanced.py   # степень, корень, факториал
└── geometry.py   # площадь круга, прямоугольника, треугольника

Пример решения:

# math_package/basic.py
def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

def multiply(a, b):
    return a * b

def divide(a, b):
    if b == 0:
        raise ValueError("Деление на ноль!")
    return a / b

# math_package/advanced.py
import math

def power(a, b):
    return a ** b

def sqrt(a):
    if a < 0:
        raise ValueError("Корень из отрицательного числа!")
    return math.sqrt(a)

def factorial(n):
    if n < 0:
        raise ValueError("Факториал отрицательного числа!")
    if n <= 1:
        return 1
    return n * factorial(n - 1)

# math_package/geometry.py
import math

def circle_area(radius):
    if radius < 0:
        raise ValueError("Радиус не может быть отрицательным")
    return math.pi * radius ** 2

def rectangle_area(width, height):
    if width < 0 or height < 0:
        raise ValueError("Стороны не могут быть отрицательными")
    return width * height

def triangle_area(base, height):
    if base < 0 or height < 0:
        raise ValueError("Основание и высота должны быть положительными")
    return 0.5 * base * height

# main.py - программа для тестирования
from math_package.basic import add, subtract, multiply, divide
from math_package.advanced import power, sqrt, factorial
from math_package.geometry import circle_area, rectangle_area, triangle_area

def main():
    print("=== Тестирование математического пакета ===")
    
    # Базовые операции
    print(f"5 + 3 = {add(5, 3)}")
    print(f"10 - 4 = {subtract(10, 4)}")
    print(f"6 * 7 = {multiply(6, 7)}")
    print(f"15 / 3 = {divide(15, 3)}")
    
    # Продвинутые операции
    print(f"2^10 = {power(2, 10)}")
    print(f"√25 = {sqrt(25)}")
    print(f"5! = {factorial(5)}")
    
    # Геометрия
    print(f"Площадь круга (r=5): {circle_area(5):.2f}")
    print(f"Площадь прямоугольника (4x6): {rectangle_area(4, 6)}")
    print(f"Площадь треугольника (основание=10, высота=5): {triangle_area(10, 5)}")

if __name__ == "__main__":
    main()
